Buffalo & Western New York

Exploring my hometown through data

library(leaflet)
dat <- data.frame()
leaflet(data=dat) %>% 
    addTiles() %>% 
    addProviderTiles(provider="HERE") %>% 
    leaflet::setView(-78.878, 42.886, zoom = 10) %>% 
  addRectangles(
    lng1=-78.6, lat1=42.7,
    lng2=-79, lat2=43.1,
    fillColor = "transparent"
  )
